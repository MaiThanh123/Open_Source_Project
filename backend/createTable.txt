CREATE TABLE fighter_stats (
    name VARCHAR(100) PRIMARY KEY,
    
    weight_class VARCHAR(50),
    nickname VARCHAR(100),
    fight_style VARCHAR(100),
    ranking INT,
    age INT,
    height_cm FLOAT,
    reach_cm FLOAT,
    country VARCHAR(100),
    stance VARCHAR(50),
    image_url VARCHAR(255),
    entry_song_url VARCHAR(255),

    wins INT,
    losses INT,
    draws INT,

    strike_acc FLOAT,
    takedown_acc FLOAT,
    strike_per_min FLOAT,
    takedown_avg_per_fight FLOAT,
    sig_strike_def FLOAT,
    takedown_def FLOAT,
    avg_fight_time_min FLOAT,

    strike INT GENERATED ALWAYS AS (strike_acc*100) VIRTUAL,
    cardio INT GENERATED ALWAYS AS (100*strike_per_min/7) VIRTUAL,
    grappling INT GENERATED ALWAYS AS (takedown_acc*100) VIRTUAL,
    power INT GENERATED ALWAYS AS (a + b) VIRTUAL,
    speed INT GENERATED ALWAYS AS (a + b) VIRTUAL,
    defense INT GENERATED ALWAYS AS (a + b) VIRTUAL,

    position_stand FLOAT,
    position_clinch FLOAT,
    position_ground FLOAT,
    
    target_head FLOAT,
    target_body FLOAT,
    target_leg FLOAT,

    win_ko_tko INT,
    win_decision INT,
    win_submission INT
);
